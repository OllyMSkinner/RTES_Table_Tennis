// This line includes the header file for the LEDController.
// It brings in the definition of the class so this file can implement it.
#include "ledcallback.hpp"


// These lines include low-level system tools for working with files and hardware.
// They allow the program to communicate directly with GPIO (pins on the Raspberry Pi).
#include <fcntl.h>
#include <linux/gpio.h>
#include <sys/ioctl.h>
#include <unistd.h>


// These provide additional utilities.
// <cstring> is used for working with text in C-style strings.
// <stdexcept> allows us to generate errors if something goes wrong.
#include <cstring>
#include <stdexcept>


// This is the constructor for LEDController.
// It runs when you create the LED object and sets up access to the GPIO pin.
LEDController::LEDController(const char* chipDevice, int gpioPin)
    : chip_fd(-1), line_fd(-1)
{
    // This opens the GPIO chip device (like opening a connection to the hardware).
    chip_fd = open(chipDevice, O_RDONLY);

    // If opening the device fails, stop the program and report an error.
    if (chip_fd < 0) {
        throw std::runtime_error("Failed to open GPIO chip device");
    }


    // This creates a request structure to ask the system for control of a GPIO pin.
    struct gpiohandle_request req{};

    // This sets which GPIO pin we want to use.
    req.lineoffsets[0] = static_cast<__u32>(gpioPin);

    // This sets the pin to output mode (we want to send signals to the LED).
    req.flags = GPIOHANDLE_REQUEST_OUTPUT;

    // This says we are only using one pin.
    req.lines = 1;

    // This gives a name to this GPIO usage (useful for debugging).
    std::strcpy(req.consumer_label, "led-controller");

    // This sets the initial value of the pin to off (0).
    req.default_values[0] = static_cast<__u8>(0);


    // This asks the system to give us control of the GPIO pin.
    if (ioctl(chip_fd, GPIO_GET_LINEHANDLE_IOCTL, &req) < 0) {

        // If it fails, close the chip and report an error.
        close(chip_fd);
        chip_fd = -1;
        throw std::runtime_error("Failed to get GPIO line handle");
    }


    // This stores the file descriptor (handle) used to control the pin later.
    line_fd = req.fd;
}


// This is the destructor.
// It runs automatically when the LEDController object is destroyed.
// It cleans up and closes any open connections to the hardware.
LEDController::~LEDController()
{
    // If the GPIO line is open, close it.
    if (line_fd >= 0) {
        close(line_fd);
    }

    // If the GPIO chip is open, close it.
    if (chip_fd >= 0) {
        close(chip_fd);
    }
}


// This function turns the LED on or off.
// It takes a true/false value: true = ON, false = OFF.
void LEDController::set(bool on)
{
    // If the GPIO line was not set up properly, throw an error.
    if (line_fd < 0) {
        throw std::runtime_error("LED GPIO line is not initialized");
    }


    // This creates a structure to hold the value we want to send to the pin.
    struct gpiohandle_data data{};

    // This sets the pin value: 1 for ON, 0 for OFF.
    data.values[0] = static_cast<__u8>(on ? 1 : 0);


    // This sends the value to the GPIO pin (actually turns LED on/off).
    if (ioctl(line_fd, GPIOHANDLE_SET_LINE_VALUES_IOCTL, &data) < 0) {

        // If it fails, report an error.
        throw std::runtime_error("Failed to set GPIO line value");
    }
}
