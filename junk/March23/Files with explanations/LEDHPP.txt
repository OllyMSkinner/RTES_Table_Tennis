#pragma once


// This defines a class called LEDController.
// A class is like a blueprint for creating something that can control an LED.
class LEDController {
public:

    // This is the constructor.
    // It sets up the LED controller when it is created.
    // It has default values:
    // "/dev/gpiochip0" is the main GPIO device on the Raspberry Pi,
    // and 22 is the pin number used for the LED.
    LEDController(const char* chipDevice = "/dev/gpiochip0", int gpioPin = 22);


    // This is the destructor.
    // It runs automatically when the object is no longer needed,
    // and it cleans up any resources being used.
    ~LEDController();


    // This function turns the LED on or off.
    // If "on" is true, the LED turns on.
    // If "on" is false, the LED turns off.
    void set(bool on);


private:

    // This stores the connection to the GPIO chip.
    // It is like a handle that allows the program to talk to the hardware.
    int chip_fd;

    // This stores the connection to the specific GPIO pin.
    // It is used to control the LED.
    int line_fd;
};
