// These lines include the files that define how position detection works
// and how the LED is controlled.
#include "positiondetector.hpp"
#include "ledcallback.hpp"


// These include tools that help with things like timing, signals,
// reading data, and working with text.
#include <atomic>
#include <chrono>
#include <csignal>
#include <cstring>
#include <fcntl.h>
#include <iostream>
#include <sstream>
#include <string>
#include <thread>
#include <unistd.h>


// This creates a shared flag that tells the program whether it should keep running.
// It starts as true, meaning the program will run until told to stop.
static std::atomic<bool> g_running{true};


// This function runs when you press Ctrl+C.
// It tells the program to stop running.
void handleSigint(int)
{
    g_running = false;
}


// This function tries to open a device (like Bluetooth) multiple times.
// If it fails, it waits and tries again.
static int open_with_retry(const char* path, int flags, int retries = 50, int ms = 100)
{
    for (int i = 0; i < retries; ++i) {

        int fd = ::open(path, flags);

        if (fd >= 0) {
            return fd;
        }

        std::this_thread::sleep_for(std::chrono::milliseconds(ms));
    }

    return -1;
}


// This is the main function where the program starts running.
int main()
{
    // This sets up Ctrl+C handling so the program can shut down safely.
    std::signal(SIGINT, handleSigint);


    // This is the path to the Bluetooth device receiving IMU data.
    const char* in_dev = "/dev/rfcomm0";


    // Try block is used to catch and handle errors safely.
    try {

        // Create the LED controller (this connects to the GPIO pin).
        LEDController led("/dev/gpiochip0", 22);

        // Create the position detector with default settings.
        PositionDetector detector(PositionDetector::Config{});


        // This tells the detector what to do when the upright state changes.
        // It prints the state and turns the LED on or off.
        detector.setCallback([&led](bool upright) {
            std::cout << "[APP] Upright state = " << upright << std::endl;
            led.set(upright);
        });


        // Try to open the Bluetooth input device.
        int fd = open_with_retry(in_dev, O_RDONLY | O_NOCTTY);

        // If it fails, print an error and stop the program.
        if (fd < 0) {
            std::cerr << "Failed to open " << in_dev << std::endl;
            return 1;
        }


        // Inform the user that the program is running.
        std::cout << "Pi 5 receiver running on " << in_dev << " (Ctrl-C to stop)\n";


        // This will store partial incoming data until full lines are formed.
        std::string pending;

        // This buffer temporarily holds raw incoming bytes.
        char buf[256];


        // This loop runs continuously until the program is stopped.
        while (g_running) {

            // Read data from the Bluetooth device.
            ssize_t n = ::read(fd, buf, sizeof(buf));

            // If reading fails, print an error and exit the loop.
            if (n < 0) {
                std::cerr << "Read failed from " << in_dev << std::endl;
                break;
            }

            // If no data is received, wait briefly and try again.
            if (n == 0) {
                std::this_thread::sleep_for(std::chrono::milliseconds(20));
                continue;
            }


            // Add the new data to the pending string.
            pending.append(buf, static_cast<std::size_t>(n));


            // Process complete lines (ending with newline).
            std::size_t pos = 0;

            while ((pos = pending.find('\n')) != std::string::npos) {

                // Extract one full line.
                std::string line = pending.substr(0, pos);

                // Remove it from the pending buffer.
                pending.erase(0, pos + 1);


                // Skip empty lines.
                if (line.empty()) {
                    continue;
                }


                // Skip header lines (like CSV column names).
                if (line.rfind("ts_ms,", 0) == 0) {
                    continue;
                }


                // Break the line into parts separated by commas.
                std::stringstream ss(line);
                std::string field;

                std::string tsStr, axStr, ayStr, azStr;
                std::getline(ss, tsStr, ',');
                std::getline(ss, axStr, ',');
                std::getline(ss, ayStr, ',');
                std::getline(ss, azStr, ',');


                // If any part is missing, ignore the line.
                if (tsStr.empty() || axStr.empty() || ayStr.empty() || azStr.empty()) {
                    continue;
                }


                // Try to convert the values into numbers.
                try {
                    float ax = std::stof(axStr);
                    float ay = std::stof(ayStr);
                    float az = std::stof(azStr);

                    // Send the values to the position detector.
                    detector.onSample(ax, ay, az);

                } catch (...) {

                    // If conversion fails, ignore the line.
                }
            }
        }


        // Close the Bluetooth connection.
        ::close(fd);

        // Turn off the LED before exiting.
        led.set(false);

        // Inform the user that the program has shut down.
        std::cout << "Shutdown complete.\n";
    }


    // If any error occurs, print the error message.
    catch (const std::exception& ex) {
        std::cerr << "Error: " << ex.what() << std::endl;
        return 1;
    }


    // End of program.
    return 0;
}
