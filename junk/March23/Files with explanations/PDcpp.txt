// This line tells the program to include another file called "positiondetector.hpp".
// Think of it like bringing in instructions or tools that this file needs to work.
#include "positiondetector.hpp"


// This is the constructor for PositionDetector.
// A constructor is something that runs when you first create the object.
// It sets up the initial state (like preparing everything before starting).
PositionDetector::PositionDetector(const Config& c)
    : cfg(c), stableCount(0), confirmed(false) {}


// This function runs every time new sensor data (ax, ay, az) is received.
// These values represent acceleration in 3 directions (like movement or tilt).
void PositionDetector::onSample(float ax, float ay, float az)
{
    // This calculates the total strength of movement (like how strong gravity or motion is).
    float g = std::sqrt(ax * ax + ay * ay + az * az);

    // This starts as false, meaning “not in the correct position yet”.
    bool accelOK = false;


    // This checks if the total movement is within an expected range.
    // If it's too small or too big, we ignore it.
    if (g >= cfg.minG && g <= cfg.maxG) {

        // These lines turn the movement into a “direction only” version (called normalizing).
        float nx = ax / g;
        float ny = ay / g;
        float nz = az / g;

        // This compares the current direction with a reference direction (what we expect).
        float dot = nx * cfg.refX + ny * cfg.refY + nz * cfg.refZ;

        // If the direction is close enough to what we expect, we say it's OK.
        accelOK = (dot >= cfg.minDot);
    }


    // If the position looks correct:
    if (accelOK) {

        // Increase the stability counter (how long it's been correct),
        // but don’t let it go above a limit.
        if (stableCount < cfg.stabilitySamples)
            ++stableCount;

    } else {

        // If it's not correct, reset the counter back to zero.
        stableCount = 0;
    }


    // This checks if we've been stable for long enough.
    bool nowConfirmed = (stableCount >= cfg.stabilitySamples);


    // If the confirmed state has changed (from false to true or true to false):
    if (nowConfirmed != confirmed) {

        // Update the stored state.
        confirmed = nowConfirmed;

        // If a callback function exists, call it and pass the new state.
        if (callback)
            callback(confirmed);
    }
}
