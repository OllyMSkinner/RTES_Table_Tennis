// This line makes sure this file is only included once in the program.
// It prevents errors from the same code being loaded multiple times.
#pragma once


// These lines bring in useful tools from the standard library.
// <functional> allows us to store and call functions.
// <cmath> gives us access to math functions like square root.
#include <functional>
#include <cmath>


// This defines a class called PositionDetector.
// A class is like a blueprint or template for creating something with data and behaviour.
class PositionDetector {
public:

    // This creates a type of function called StateCallback.
    // It represents a function that takes a true/false value (upright or not)
    // and does something with it.
    using StateCallback = std::function<void(bool upright)>;


    // This structure holds configuration values.
    // Think of it as a box of settings that control how the detector behaves.
    struct Config {

        // These define the “correct” direction the device should be pointing.
        float refX = -0.81f;
        float refY = -0.56f;
        float refZ =  0.20f;

        // This controls how close the current direction must be to the correct one.
        float minDot = 0.65f;

        // This is how many correct readings in a row are needed before confirming.
        int stabilitySamples = 4;

        // These define the acceptable strength of movement (like gravity range).
        float minG = 8.0f;
        float maxG = 11.0f;
    };


    // This creates a PositionDetector with default settings.
    PositionDetector();


    // This creates a PositionDetector with custom settings provided by the user.
    explicit PositionDetector(const Config& cfg);


    // This function is called whenever new sensor data is received.
    // It processes acceleration in X, Y, and Z directions.
    void onSample(float ax, float ay, float az);


    // This function allows you to give the detector a function to call
    // whenever the upright state changes.
    void setCallback(StateCallback cb) { callback = std::move(cb); }


private:

    // This stores the configuration settings being used.
    Config cfg;

    // This keeps track of how many correct readings in a row we’ve had.
    int stableCount = 0;

    // This stores whether the device is currently confirmed upright.
    bool confirmed = false;

    // This stores the function that gets called when the state changes.
    StateCallback callback;
};
